            try {

        async function loadDashboard() {
            try {
                // Busca dados reais do Supabase - USANDO NOMES PLURAIS CORRETOS
                const { count: countLeads, error: err1 } = await supabase.from('leads').select('*', { count: 'exact', head: true });
                const { count: countAlunos, error: err2 } = await supabase.from('alunos').select('*', { count: 'exact', head: true }).eq('status', 'Ativo');
                const { data: receitaData, error: err3 } = await supabase.from('financeiro').select('valor').eq('status', 'Pago');
                const { data: inadimplenciaData, error: err4 } = await supabase.from('financeiro').select('valor').eq('status', 'Atrasado');
                
                if (err1) throw err1;
                if (err2) throw err2;
                
                const receitaTotal = receitaData ? receitaData.reduce((sum, item) => sum + Number(item.valor), 0) : 0;
                const inadimplenciaTotal = inadimplenciaData ? inadimplenciaData.reduce((sum, item) => sum + Number(item.valor), 0) : 0;
                
                document.getElementById('dash-total-leads').textContent = countLeads || 0;
                document.getElementById('dash-total-alunos').textContent = countAlunos || 0;
                document.getElementById('dash-receita').textContent = 'R$ ' + receitaTotal.toFixed(2).replace('.', ',');
                document.getElementById('dash-inadimplencia').textContent = 'R$ ' + inadimplenciaTotal.toFixed(2).replace('.', ',');
            } catch (error) {
                console.error('Erro no dashboard:', error);
            }
        }

        async function loadTable(table, elementId) {
            const { data, error } = await supabase.from(table).select('*').order('created_at', { ascending: false });
            if (error) throw error;
            
            const tbody = document.getElementById(elementId);
            tbody.innerHTML = '';
            
            if (!data || data.length === 0) {
                tbody.innerHTML = '<tr><td colspan="6" class="p-4 text-center text-gray-500">Nenhum registro encontrado</td></tr>';
                return;
            }
            
            data.forEach(row => {
                const tr = document.createElement('tr');
                tr.className = 'border-b hover:bg-gray-50 transition';
                
                let cells = '';
                if (table === 'leads') {
                    cells = `<td class="p-3 font-medium">${row.nome}</td><td class="p-3">${row.email || '-'}</td><td class="p-3">${row.telefone || '-'}</td><td class="p-3"><span class="px-2 py-1 bg-blue-100 rounded text-sm">${row.etapa}</span></td>`;
                } else if (table === 'alunos') {
                    cells = `<td class="p-3 font-medium">${row.nome}</td><td class="p-3">${row.email || '-'}</td><td class="p-3">${row.telefone || '-'}</td><td class="p-3">${row.nivel}</td><td class="p-3"><span class="px-2 py-1 bg-green-100 rounded text-sm">${row.status}</span></td>`;
                }
                
                tr.innerHTML = cells + `
                    <td class="p-3">
                        <button onclick="deleteItem('${table}', '${row.id}')" class="text-red-600 hover:text-red-800 transition"><i class="fas fa-trash"></i></button>
                    </td>
                `;
                tbody.appendChild(tr);
            });
        }

        window.saveData = async function(e) {
            e.preventDefault();
            const form = e.target;
            const formData = new FormData(form);
            const data = Object.fromEntries(formData.entries());
            
            try {
                const { error } = await supabase.from(currentTable).insert(data);
                
                if (error) {
                    Swal.fire('Erro', error.message, 'error');
                } else {
                    Swal.fire('Sucesso', 'Registro salvo com sucesso!', 'success');
                    closeModal();
                    loadData(currentTable);
                    loadDashboard(); // Atualiza dashboard após salvar
                }
            } catch (error) {
                Swal.fire('Erro', 'Erro ao salvar: ' + error.message, 'error');
            }
        };

        window.deleteItem = async function(table, id) {
            if (!confirm('Confirmar exclusão?')) return;
            try {
                const { error } = await supabase.from(table).delete().eq('id', id);
                if (error) throw error;
                Swal.fire('Excluído!', 'Registro removido com sucesso.', 'success');
                loadData(table);
                loadDashboard();
            } catch (error) {
                Swal.fire('Erro', error.message, 'error');
            }
        };

        window.markPaid = async function(id) {
            try {
                const { error } = await supabase.from('financeiro').update({ 
                    status: 'Pago', 
                    data_pagamento: new Date().toISOString() 
                }).eq('id', id);
                
                if (error) throw error;
                Swal.fire('Pago!', 'Cobrança marcada como paga.', 'success');
                loadFinanceiro();
                loadDashboard();
            } catch (error) {
                Swal.fire('Erro', error.message, 'error');
            }
        };

        // Inicializa o sistema
        showPage('dashboard');
